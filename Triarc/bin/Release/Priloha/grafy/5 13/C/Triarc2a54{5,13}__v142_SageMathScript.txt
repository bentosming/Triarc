    return x*(n-j)

M=matrix(QQ,n,n,lambda i,j: scaling(Munscaled[i,j],i,j,n) if i>j & i<countOfVertices else Munscaled[i,j])

    M[i+outerSize,i+outerSize]+=soucty[i+outerSize,0]
